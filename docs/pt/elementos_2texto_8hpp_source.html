98 // retorna a largura da linha mais longa dentro de 'frase'
99 float obterLargura(const std::string& frase) {
100 if (m_texto_frase_ptr) m_texto_frase = *m_texto_frase_ptr;
101 auto& caracteres = gerenciadorFontes::obterInstancia()
102 .obter(m_texto_fonte, m_texto_escala);
103 if (caracteres.empty()) {
104 return 0.0f;
105 }
106
107 float maxLargura = 0.0f;
108 float larguraLinha = 0.0f;
109
110 for (char32_t c : frase) {
111 if (c == '\n') {
112 // finaliza a linha atual, atualiza o máximo e reseta
113 maxLargura = std::max(maxLargura, larguraLinha);
114 larguraLinha = 0.0f;
115 continue;
116 }
117 auto it = caracteres.find(c);
118 if (it != caracteres.end()) {
119 const caractere& ch = it->second;
120 larguraLinha += (ch.avanco >> 6);
121 }
122 }
123 // compara a última linha (caso não termine em '\n')
124 maxLargura = std::max(maxLargura, larguraLinha);
125 return maxLargura;
126 }

128 // retorna a altura total considerando quantas linhas existem em 'frase'
129 float obterAltura(const std::string& frase) {
130 if (m_texto_frase_ptr) m_texto_frase = *m_texto_frase_ptr;
131 auto& caracteres = gerenciadorFontes::obterInstancia()
132 .obter(m_texto_fonte, m_texto_escala);
133 if (caracteres.empty()) {
134 return 0.0f;
135 }
136
137 // conta quantas quebras de linha existem → número de linhas = quebras + 1
138 size_t numQuebras = 0;
139 for (char32_t c : frase) {
140 if (c == '\n') {
141 ++numQuebras;
142 }
143 }
144 size_t numLinhas = numQuebras + 1;
145
146 // precisa de um método no gerenciador para expor a "altura de linha"
147 // por exemplo: gerenciadorFontes::obterAlturaLinha(fonte, escala)
148 float alturaLinha = m_texto_escala * 1.4;
149
150 return alturaLinha * static_cast<float>(numLinhas);
151 }

153 limites_iniciais = m_estilo.m_limites;
154
155 auto& chs = gerenciadorFontes::obterInstancia().obter(m_texto_fonte, m_texto_escala);
156 float y_linha = m_texto_escala;

159 x_linha += m_estilo.m_limites.z / 2 - obterLargura(m_texto_frase)/2;

173 else
