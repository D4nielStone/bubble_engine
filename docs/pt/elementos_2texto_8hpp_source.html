101 // retorna a largura da linha mais longa dentro de 'frase'
102 float obterLargura(const std::string& frase) {
103 if (m_texto_frase_ptr) m_texto_frase = *m_texto_frase_ptr;
104 auto& caracteres = gerenciadorFontes::obterInstancia()
105 .obter(m_texto_fonte, m_texto_escala);
106 if (caracteres.empty()) {
107 return 0.0f;
108 }
109
110 float maxLargura = 0.0f;
111 float larguraLinha = 0.0f;
112
113 for (char32_t c : frase) {
114 if (c == '\n') {
115 // finaliza a linha atual, atualiza o máximo e reseta
116 maxLargura = std::max(maxLargura, larguraLinha);
117 larguraLinha = 0.0f;
118 continue;
119 }
120 auto it = caracteres.find(c);
121 if (it != caracteres.end()) {
122 const caractere& ch = it->second;
123 larguraLinha += (ch.avanco >> 6);
124 }
125 }
126 // compara a última linha (caso não termine em '\n')
127 maxLargura = std::max(maxLargura, larguraLinha);
128 return maxLargura;
129 }

131 // retorna a altura total considerando quantas linhas existem em 'frase'
132 float obterAltura(const std::string& frase) {
133 if (m_texto_frase_ptr) m_texto_frase = *m_texto_frase_ptr;
134 auto& caracteres = gerenciadorFontes::obterInstancia()
135 .obter(m_texto_fonte, m_texto_escala);
136 if (caracteres.empty()) {
137 return 0.0f;
138 }
139
140 // conta quantas quebras de linha existem → número de linhas = quebras + 1
141 size_t numQuebras = 0;
142 for (char32_t c : frase) {
143 if (c == '\n') {
144 ++numQuebras;
145 }
146 }
147 size_t numLinhas = numQuebras + 1;
148
149 // precisa de um método no gerenciador para expor a "altura de linha"
150 // por exemplo: gerenciadorFontes::obterAlturaLinha(fonte, escala)
151 float alturaLinha = m_texto_escala * 1.4;
152
153 return alturaLinha * static_cast<float>(numLinhas);
154 }
